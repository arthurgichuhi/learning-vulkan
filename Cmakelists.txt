cmake_minimum_required(VERSION 3.10)
project(learning-vulkan)

set(CMAKE_EXPORT_COMPILE_COMMANDS ON)
set(CMAKE_CXX_STANDARD 17)


# Find packages
find_package(Vulkan REQUIRED)
# GLM
set(GLM_INCLUDE_DIR ${CMAKE_CURRENT_SOURCE_DIR}/libs/glm)

# Add glm headers
include_directories(${GLM_INCLUDE_DIR})

# GLFW
set(GLFW_DIR ${CMAKE_SOURCE_DIR}/libs/glfw)

set(GLFW3_INCLUDE_DIRS ${GLFW_DIR}/include)
set(GLFW3_LIBRARY_DIRS ${GLFW_DIR}/lib)

# Find the library file
find_library(GLFW3_LIBRARY 
    NAMES glfw3_mt glfw3dll glfw3
    PATHS ${GLFW3_LIBRARY_DIRS}
    REQUIRED
)

# Create imported target
add_library(glfw INTERFACE)
target_include_directories(glfw INTERFACE ${GLFW3_INCLUDE_DIRS})
target_link_libraries(glfw INTERFACE ${GLFW3_LIBRARY})


add_executable(${PROJECT_NAME} 
src/main.cpp
src/core/engine.cpp
src/logging.cpp
src/core/scene.cpp
src/app.cpp
 "src/shaders.h" 
 "src/pipeline.h" 
 "src/framebuffer.h" 
 "src/commands.h"  
 "src/frame.h" "src/sync.h" 
 "src/render_structs.h")
target_include_directories(${PROJECT_NAME} PRIVATE ${GLFW3_INCLUDE_DIRS})
target_link_libraries(${PROJECT_NAME} PRIVATE 
    ${GLFW3_LIBRARY}
    Vulkan::Vulkan
)
